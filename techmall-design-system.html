📐 Vue3 / Vite 使用指南

INTEGRATION
/* ============================================================ 步骤 1: 全局 CSS 变量 — src/styles/variables.css 复制上方 CSS 变量速查表全部内容至此文件 ============================================================ */ /* ============================================================ 步骤 2: 全局样式 — src/styles/global.css ============================================================ */ @import './variables.css'; *, *::before, *::after { margin: 0; padding: 0; box-sizing: border-box; } html { font-size: 16px; } body { font-family: var(--font-body); background: var(--bg-deep); color: var(--text-primary); -webkit-font-smoothing: antialiased; line-height: 1.6; } a { color: var(--accent-cyan); text-decoration: none; } /* ============================================================ 步骤 3: main.ts 中引入 ============================================================ */ import '@/styles/global.css' /* ============================================================ 步骤 4: Element Plus 暗色覆盖 (如果使用) ============================================================ */ // 在 vite.config.ts 中配置 SCSS 变量覆盖 Element Plus 默认主题 // 或者直接使用 CSS 变量覆盖常用组件 :root { --el-bg-color: var(--bg-surface); --el-bg-color-overlay: var(--bg-elevated); --el-text-color-primary: var(--text-primary); --el-text-color-regular: var(--text-secondary); --el-border-color: var(--border-subtle); --el-color-primary: var(--accent-cyan); --el-border-radius-base: var(--radius-sm); } /* ============================================================ 步骤 5: 常用工具类 ============================================================ */ .text-cyan { color: var(--accent-cyan); } .text-amber { color: var(--accent-amber); } .text-muted { color: var(--text-muted); } .font-display { font-family: var(--font-display); } .glass { background: var(--bg-glass); backdrop-filter: blur(20px) saturate(180%); } .card { background: var(--bg-surface); border: 1px solid var(--border-subtle); border-radius: var(--radius-md); }
